Clean

Change datetime_arrest to datetime object. Get rid of columns officer, offense_status, case_status.

pecos_clean <- pecos |> mutate(
  datetime_arrest = mdy_hm(datetime_arrest)
) |> select(
  -officer,
  -offense_status,
  -case_status
)
