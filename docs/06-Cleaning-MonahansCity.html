Clean

Change date_arrest to a date type and split race_sex column into two.

monahans_clean <- monahans |> mutate(
  race = substr(race_sex, 1, 1),
  sex = substr(race_sex, 3, 3),
  date_arrest = mdy(date_arrest)
  ) |> select( -race_sex)
